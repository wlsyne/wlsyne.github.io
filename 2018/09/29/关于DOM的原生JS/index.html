原生JS操作DOM

1.获取高度和滚动距离

<!-- 获取滚动距离 -->
document.documentElement.scrollTop //firefox

document.documentElement.scrollLeft //firefox

document.body.scrollTop //IE

document.body.scrollLeft //IE

<!-- 获取窗口高度 -->
document.documentElement.clientHeight

<!-- 获取整个文档的高度 -->
document.documentElement.scrollHeight||document.body.scrollHeight

2.设置css和获取css

element.style.textAlign=center
<!-- 如果要设置!important -->
element.style.setProperty('height', '300px', 'important');
<!-- 获取css -->
function Getstyle(ele) {
        let style = window.getComputedStyle
            ? window.getComputedStyle(ele, "")
            : ele.currentStyle;
        return style
    }

3.Document

Document接口表示任何在浏览器中已经加载好的网页，并作为一个入口去操作网页内容（也就是DOM tree）

<!-- 获取body -->
Document.body

<!-- 获取html -->
Document.documentElement

<!-- 获取/设置title -->
Document.title
Document.title=''

<!-- 创建一个html元素 -->
Document.createElement('div')

<!-- 创建一个文本节点 -->
Document.createTextNode('春江潮水连海平')

<!-- 通过class来查找元素 -->
let elements = document.getElementsByClassName(names); 
let elements = rootElement.getElementsByClassName(names);
//这个api可以在任何元素上调用，调用这个api的元素将作为本次查找的根元素
//返回包含这个class类名的NodeList

<!-- 通过标签名称来查找元素 -->
let elements = document.getElementsByTagName(names); 
let elements = rootElement.getElementsByTagName(names);
//这个api可以在任何元素上调用，调用这个api的元素将作为本次查找的根元素
//返回符合这个标签名称的NodeList

<!-- 通过id来查找元素 -->
let elements = document.getElementById(names); 
//这个api只能够在Document上调用
//返回拥有指定id的element，如果没有找到则返回null

<!-- 通过选择器查找单个元素 -->
let elements = document.querySelector(selectors); 
let elements = rootElement.querySelector(selectors);
//这个api可以在任何元素上调用，调用这个api的元素将作为本次查找的根元素
//返回文档中与指定选择器或选择器组匹配的第一个元素，如果找不到匹配项，则返回null。

<!-- 通过选择器查找元素 -->
let elements = document.querySelectorAll(selectors); 
let elements = rootElement.querySelectorAll(selectors);
//这个api可以在任何元素上调用，调用这个api的元素将作为本次查找的根元素
//返回的对象是 NodeList

4.Element

Element是非常通用的基类，所有Document对象下的对象都继承它都继承它. 这个接口描述了所有相同种类的元素所普遍具有的方法和属性
同时Element也是Node对象的一个类型，继承于Node对象

属性

获取/设置元素的class

let clas=Element.classList
let clas=Element.className
Element.className='wlsyne'
//classList 返回的是一个DOMTokenList,这个api是只读的
//className返回的是一个string

获取元素的宽高

let height=Element.clientHeight
let height=Element.offsetHeight

let width=Element.clientWidth
let width=Element.offsetWidth
//offset包括边框 client包括边框

获取/设置元素的id

let id=Element.id
id='wlsyne'

获取/设置元素内部的html

let content=Element.innerHTML
content='<div>嘻嘻嘻嘻</div>'

获取/设置元素的html（包括该元素）

let content=Element.outerHTML
content='<div>嘻嘻嘻嘻</div>'

元素滚动相关

<!-- 滚动视图高度 -->
let scrollHeight=Element.scrollHeight
<!-- 纵向滚动距离 -->
let scrollTop=Element.scrollTop

获取元素的标签名

let name=Element.tagName
方法

匹配特定选择起且离当前元素最近的祖先元素

let elt = element.closest(selectors); 
<!-- 一个polyfill -->
  function closest(el,selector){
       let matchesSelector = el.matches || el.webkitMatchesSelector || el.mozMatchesSelector || el.msMatchesSelector;
        while (el) {
            if (matchesSelector.call(el, selector)) {
                break;
            }
            el = el.parentElement;
        }
  }

属性相关

<!-- 获取元素的指定属性 -->
let attribute = element.getAttribute(attributeName);
<!-- 移除指定属性 -->
element.removeAttribute(attrName)
<!-- 设定属性 -->
element.setAttribute(name, value);

获取元素的大小及其相对于视口的位置

let rect = element.getBoundingClientRect();
//位置是相对于视口左上角而言

判断元素是否含有相关属性

let result = element.hasAttribute(attName);

判断元素被指定的选择器字符串选择

let result = element.matches(selectorString);
//一个polyfill
if (!Element.prototype.matches) {
    Element.prototype.matches = 
        Element.prototype.matchesSelector || 
        Element.prototype.mozMatchesSelector ||
        Element.prototype.msMatchesSelector || 
        Element.prototype.oMatchesSelector || 
        Element.prototype.webkitMatchesSelector ||
        function(s) {
            var matches = (this.document || this.ownerDocument).querySelectorAll(s),
                i = matches.length;
            while (--i >= 0 && matches.item(i) !== this) {}
            return i > -1;            
        };
}

5.Node

Node是一个接口，许多DOM类型从这个接口继承。
以下接口都从Node继承其方法和属性：
Document, Element, CharacterData (which Text, Comment, and CDATASection inherit), ProcessingInstruction, DocumentFragment, DocumentType, Notation, Entity, EntityReference

属性

子节点相关

<!-- 获取所有子节点（包含所有的节点类型：text等） -->
var ndList = elementNodeReference.childNodes;
<!-- 获取所有子节点（只包含Element节点） -->
let children = node.children;
<!-- 获取第一个子节点 -->
let firstNode=node.firstChild;//可能是text节点
let firstElem=node.firstElementChild;//只能是Element节点
<!-- 获取最后一个子节点 -->
let lastNode=node.lastChild;//可能是text节点
let lastElem=node.lastElementChild;//只能是Element节点

同级节点相关

<!-- 获取下一个同级节点 -->
let nextNode=node.nextSibling
let nextElem=node.nextElementSibling//只能是Element
<!-- 获取上一个同级节点 -->
let preNode=node.previousSibling
let preElem=node.previousElementSibling//只能是Element

父节点相关

let parent=node.parentNode
let parentElem=node.parentElement

获取节点名称

let name=node.nodeName
//nodeName获取的值都是大写

获取节点类型

let name=node.nodeType
//1.Element_Node
//2.Attribute_Node
//3.Text_Node
方法

子节点相关

<!-- 向父节点末尾增加节点 -->
let child = node.appendChild(child);
<!-- 向指定子节点之前插入节点 -->
let insertedNode = parentNode.insertBefore(newNode, referenceNode);
<!-- 替换子节点 -->
let replacedNode = parentNode.replaceChild(newChild, oldChild);
<!-- 移除子节点 -->
let oldChild = node.removeChild(child);

克隆节点

var dupNode = node.cloneNode(deep);
//如果deep是true的话则会拷贝子节点

判断一个节点是否包含指定节点

node.contains( otherNode )

判断两个节点是否相等

node.isEqualNode(otherNode);
